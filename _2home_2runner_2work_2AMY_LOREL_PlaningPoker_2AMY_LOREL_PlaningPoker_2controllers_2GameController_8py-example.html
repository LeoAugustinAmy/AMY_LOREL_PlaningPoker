100 def calculate_result(self):
101 """!
102 @brief Calcule le résultat du vote selon les règles (Unanimité au T1, puis règle choisie).
103 @return Résultat numérique/str, "?" si inconnu, ou None si pas de décision.
104 @note Ignore les cartes "interro" et "cafe" pour le calcul.
105 """
106 votes_values = [v for v in self.model.votes.values() if v not in ["interro", "cafe"]]
107
108 numeric_votes = []
109 for v in votes_values:
110 if v.isdigit(): numeric_votes.append(int(v))
111
112 if self.model.current_round_number == 1:
113 if not numeric_votes: return None
114 if len(set(numeric_votes)) == 1:
115 return numeric_votes[0]
116 else:
117 return None
118
119 if not numeric_votes: return "?"
120
121 mode = self.model.rules.selected_mode
122
123 if mode == "Unanimité":
124 return numeric_votes[0] if len(set(numeric_votes)) == 1 else None
125 elif mode == "Moyenne":
126 return round(mean(numeric_votes))
127 elif mode == "Médiane":
128 return round(median(numeric_votes))
129 elif mode == "Majorité Absolue":
130 counts = {x: numeric_votes.count(x) for x in set(numeric_votes)}
131 winner = max(counts, key=counts.get)
132 if counts[winner] > len(numeric_votes) / 2:
133 return winner
134 return None
135 elif mode == "Majorité Relative":
136 res = multimode(numeric_votes)
137 return max(res)
138
139 return None

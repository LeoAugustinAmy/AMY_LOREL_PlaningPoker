258 def _get_card_image(self, value, size):
259 """!
260 @brief Charge et retourne l'image CTkImage d'une carte donn√©e.
261 @param value Valeur de la carte (ex: '20', 'cafe').
262 @param size Tuple (largeur, hauteur).
263 @return L'objet CTkImage correspondant.
264 @note Retourne None si l'image SVG est introuvable ou invalide.
265 """
266 cache_key = (value, size)
267 if cache_key in self._card_image_cache:
268 return self._card_image_cache[cache_key]
269
270 filename = f"cartes_{value}.svg"
271 image_path = os.path.abspath(os.path.join(os.path.dirname(__file__), "..", "src", "img", filename))
272
273 try:
274 with open(image_path, "rb") as f:
275 svg_data = f.read()
276 png_data = cairosvg.svg2png(bytestring=svg_data, output_height=size[1]*2)
277 pil_image = Image.open(io.BytesIO(png_data))
278 ctk_image = ctk.CTkImage(light_image=pil_image, dark_image=pil_image, size=size)
279 self._card_image_cache[cache_key] = ctk_image
280 return ctk_image
281 except Exception as e:
282 print(f"Erreur chargement image {filename}: {e}")
283 return None

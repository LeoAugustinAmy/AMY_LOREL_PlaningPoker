56 def load_game(self):
57 """!
58 @brief Charge une partie depuis un fichier JSON et redirige vers la bonne vue.
59 @details Si la partie est FINISHED -> ResultView. Sinon -> GameView.
60 @raises OSError Si la lecture du fichier échoue.
61 @raises ValueError Si le contenu JSON est invalide pour GameSession.
62 """
63 filename = filedialog.askopenfilename(filetypes=[("JSON files", "*.json")])
64 if not filename:
65 return
66
67 try:
68 with open(filename, 'r', encoding='utf-8') as f:
69 data = json.load(f)
70
71 self.game_session.from_dict(data)
72 self.game_controller.reset()
73
74 status = data.get("status", "IN_PROGRESS")
75
76 if status == "FINISHED":
77 CustomPopup("Chargement", "Partie terminée chargée. Affichage des résultats.", type="info")
78 self.show_result()
79
80 else:
81 if self.game_session.get_current_feature():
82 CustomPopup("Chargement", "Partie reprise avec succès !", type="info")
83 self.show_game()
84 else:
85 self.show_result()
86
87 except Exception as e:
88 CustomPopup("Erreur", f"Impossible de charger la partie :\n{e}", type="error")
